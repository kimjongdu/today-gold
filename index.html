async function fetchMetalPrices() { try { const apiUrl = "https://api.coingecko.com/api/v3/simple/price?ids=gold,silver,platinum&vs_currencies=krw&include_24hr_change=true"; const response = await fetch(apiUrl); if (!response.ok) { throw new Error(`HTTP 오류 발생: ${response.status}`); } const data = await response.json(); console.log("API 응답 데이터:", data); // 🟢 API 응답 로그 확인 // ✅ 응답 데이터 검증 if (!data || Object.keys(data).length === 0) { throw new Error("API 응답이 비어 있습니다."); } // ✅ 금(Gold) 변환 (트로이 온스 -> 1g 가격 변환) const goldPrice = data.gold?.krw > 0 ? (1 / data.gold.krw) / 31.1035 : 92000; // ✅ 은(Silver) 변환 (온스 기준 가격을 1g으로 변환) let silverPrice = data.silver?.krw > 0 ? (1 / data.silver.krw) / 31.1035 : 1100; // ✅ 백금(Platinum) 값이 없는 경우 기본값 설정 const platinumPrice = data.platinum?.krw > 0 ? (1 / data.platinum.krw) : 35000; // ✅ 변환된 금속 가격 const metalPrices = { gold: goldPrice, silver: silverPrice, platinum: platinumPrice }; console.log("보정된 금속 가격:", metalPrices); renderGoldRates(metalPrices); } catch (error) { console.error("API 호출 오류:", error); document.getElementById("gold-rates").innerHTML = "
데이터를 불러올 수 없습니다.
"; } }
